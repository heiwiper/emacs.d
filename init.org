#+begin_src emacs-lisp
    (defun heiwiper/count-inbox-tasks ()
    "Count top-level Org headings (* ...) in ~/org/inbox.org."
    (let ((inbox-file "~/org/inbox.org"))
      (if (file-exists-p inbox-file)
          (with-temp-buffer
            (insert-file-contents-literally inbox-file)
            (goto-char (point-min))
            (let ((count 0))
              (while (re-search-forward "^\\* " nil t)
                (cl-incf count))
              count))
        0)))

  (defvar my-captured-tasks-count 0
    "Current count of tasks under 'Captured Tasks 📸' heading.")

  (defun my-count-captured-tasks ()
    "Count top-level headings under 'Captured Tasks 📸' in ~/org/inbox.org."
    (let ((inbox-file "~/org/inbox.org"))
      (if (file-exists-p inbox-file)
          (with-temp-buffer
            (insert-file-contents-literally inbox-file)
            (org-mode)
            (goto-char (point-min))
            (if (re-search-forward (rx bol "* " "Captured Tasks 📸" eol) nil t)
                (let ((level (org-outline-level))
                      (count 0))
                  (while (org-get-next-sibling)
                    (when (= (org-outline-level) (1+ level))
                      (cl-incf count)))
                  count)
              0))
        0)))

  (defun my-update-captured-tasks-count ()
    "Update the captured tasks count and refresh mode line."
    (interactive)
    (setq my-captured-tasks-count (heiwiper/count-inbox-tasks))
    (force-mode-line-update))

  (add-hook 'org-capture-after-finalize-hook #'my-update-captured-tasks-count)
  (add-hook 'org-agenda-finalize-hook #'my-update-captured-tasks-count)
  (run-with-idle-timer 300 t #'my-update-captured-tasks-count)

  (my-update-captured-tasks-count)

  (add-to-list 'mode-line-misc-info
               '(:eval (unless (zerop my-captured-tasks-count)
                          (format "📥 %d" my-captured-tasks-count))))

  (defun lsp-booster--advice-json-parse (old-fn &rest args)
    "Try to parse bytecode instead of json."
    (or
     (when (equal (following-char) ?#)
       (let ((bytecode (read (current-buffer))))
         (when (byte-code-function-p bytecode)
           (funcall bytecode))))
     (apply old-fn args)))
  (advice-add (if (progn (require 'json)
                         (fboundp 'json-parse-buffer))
                  'json-parse-buffer
                'json-read)
              :around
              #'lsp-booster--advice-json-parse)

  (defun lsp-booster--advice-final-command (old-fn cmd &optional test?)
    "Prepend emacs-lsp-booster command to lsp CMD."
    (let ((orig-result (funcall old-fn cmd test?)))
      (if (and (not test?)
               (not (file-remote-p default-directory))
               lsp-use-plists
               (not (functionp 'json-rpc-connection))
               (executable-find "emacs-lsp-booster"))
          (progn
            (when-let ((command-from-exec-path (executable-find (car orig-result))))
              (setcar orig-result command-from-exec-path))
            (message "Using emacs-lsp-booster for %s!" orig-result)
            (cons "emacs-lsp-booster" orig-result))
        orig-result)))
  (advice-add 'lsp-resolve-final-command :around #'lsp-booster--advice-final-command)
